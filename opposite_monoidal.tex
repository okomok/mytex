% opposite enriched category

\strdpreview{
    \strdgathered{
        \strdsidelabeled{line}{\strdorigin}{A}{B}
        \strdstartlabel{AB}{\mathcal{A}^{\operatorname{op}}}
    } \strddefeq
    \strdgathered{
        \strdsidelabeled{line}{\strdorigin}{B}{A}
        \strdstartlabel{BA}{\mathcal{A}}
    }
}

\strdpreview{
    \strdgathered{
        \strdsidelabeled{line}{\strdorigin}{A}{C}
        \strdstartlabel{AC}{\mathcal{A}^{\operatorname{op}}}
        \strdsidelabeled{left}{AC}{A}{B}
        \strdsidelabeled{right}{AC}{B}{C}
        \strdjunc{AC End}
    } \strddefeq
    \strdgathered{
        \strdsidelabeled{line}{\strdorigin}{C}{A}
        \strdstartlabel{CA}{\mathcal{A}}
        \strdnolabels{\strdsidelabeled{left}{CA}{C}{B}}
        \strdnolabels{\strdsidelabeled{right}{CA}{B}{A}}
        \strdjunc{CA End}
        \strdunleft{CB}{CB2} \strdsidelabel{CB2 Start}{C}{B}
        \strdunright{BA}{BA2} \strdsidelabel{BA2 Start}{B}{A}
        \strdright{CB2}{CB3} \strdsidelabel{CB3}{C}{B}
        \strdleft{BA2}{BA3} \strdsidelabel{BA3}{B}{A}
    }
}
