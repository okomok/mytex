% composition-compatibility
\myin[preview,math]{
        \strdin[gathered]{
            \strdmid{\strdorigin}{fc} \strdsidelabel{fc}{}{FC}
            \strdmid{fc}{c} \strdsidelabel{c}{}{C}
            \strdcelllabel{c End}{g}
            \strdmid{c}{b} \strdsidelabel{b}{}{B}
            \strdcelllabel{b End}{f}
            \strdmid{b}{a} \strdsidelabel{a}{}{A}
            \strdmid{a}{fa} \strdsidelabel{fa}{}{FA}
            \strdscale{1}{3}{\strdmapsq{fc}{b}}
            \strdfamilylabel{b Right}{}{F}
        } =
        \strdin[gathered]{
            \strdmid{\strdorigin}{fc} \strdsidelabel{fc}{}{FC}
            \strdmid{fc}{c} \strdsidelabel{c}{}{C}
            \strdcelllabel{c End}{g}
            \strdmid{c}{b} \strdsidelabel{b}{}{B}
            \strdmid{b}{fb} \strdsidelabel{fb}{}{FB}
            \strdmid{fb}{b2} \strdsidelabel{b2}{}{B}
            \strdcelllabel{b2 End}{f}
            \strdmid{b2}{a} \strdsidelabel{a}{}{A}
            \strdmid{a}{fa} \strdsidelabel{fa}{}{FA}
            \strdscale{1}{2}{\strdmapsq{fc}{b}}
            \strdscale{1}{2}{\strdmapsq{fb}{b2}}
            \strdfamilylabel{b Right}{}{F}
            \strdfamilylabel{b2 Right}{}{F}
        }
}

% naturality of phi

        \strdin[gathered]{
            \strdxscale{1.5}{ \strdbranch{\strdorigin}{fab} } \strdsidelabel{fab}{}{F(A\myprime \otimes B\myprime)}
            \strdcelllabel{fab End}{\phi}
            \strdsidelabel{fab Left}{}{FB\myprime}
            \strdsidelabel{fab Right}{}{FA\myprime}
            \strdmid{fab Left}{b'} \strdsidelabel{b'}{}{B'}
            \strdcelllabel{b' End}{g}
            \strdmid{b'}{b} \strdsidelabel{b}{}{B}
            \strdmid{b}{fb} \strdsidelabel{fb}{}{FB}
            \strdyscale{2}{\strdmapsq{fab Left}{gf}} \strdfamilylabel{gf Right}{}{F}
            \strdmid{fab Right}{a'} \strdsidelabel{a'}{}{A'}
            \strdcelllabel{a' End}{f}
            \strdmid{a'}{a} \strdsidelabel{a}{}{A}
            \strdmid{a}{fa} \strdsidelabel{fa}{}{FA}
            \strdyscale{2}{\strdmapsq{fab Right}{ff}} \strdfamilylabel{ff Right}{}{F}
        } =
        \strdin[gathered]{
            \strdmid{\strdorigin}{fab}  \strdsidelabel{fab}{}{F(A\myprime \otimes B\myprime)}
            \strdbranch{fab}{ab} \strdsidelabel{ab}{}{A\myprime \otimes B\myprime}
 %           \strdcelllabel{ab End}{=}
            \strdscale{2}{3}{ \strdmapsq{fab}{f} } \strdfamilylabel{f Right}{}{F}
            \strdsidelabel{ab Left}{}{B\myprime}
            \strdsidelabel{ab Right}{}{A\myprime}
            \strdmid{ab Left}{b} \strdsidelabel{b}{}{B}
            \strdcelllabel{b Start}{g}
            \strdmid{b}{fb} \strdsidelabel{fb}{}{FB}
            \strdmid{ab Right}{a} \strdsidelabel{a}{}{A}
            \strdcelllabel{a Start}{f}
            \strdmid{a}{fa} \strdsidelabel{fa}{}{FA}
        }
}
