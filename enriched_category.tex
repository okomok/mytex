\strdpreview {
    \strdgathered {
        \strdmid{0,0}{ac}
        \strdjunc{ac To}{m}
        \strdleft{m}{ab}
        \strdright{m}{bc}
        \strdstart{ac From}{\mathcal{A}}
        \strdhomobj{ac}{A}{C}
        \strdhomobj{ab}{A}{B}
        \strdhomobj{bc}{B}{C}
    }
    \coloneqq   
    \strdgathered {
        \strdmid{0,0}{ac}
        \strdtwocell{ac To}{m}{M}
        \strdxscale{1.2}{ \strdleft{m}{ab} }
        \strdxscale{1.2}{ \strdright{m}{bc} }
        \strdstart{ac From}{\mathcal{A}(A,C)}
        \strdend{ab To}{\mathcal{A}(A,B)}
        \strdend{bc To}{\mathcal{A}(B,C)}
    }
}
