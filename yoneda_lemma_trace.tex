    } \mydefeq

        \strdcellnode{v Left Start}{1}

    \strdin[gathered]{
        \strdroad{mid}{\strdorigin}{u} \strdroadlabel{u}{}{a}
        \strddeadend{u}
    } \\
    & \myitem
    \strdin[gathered]{
